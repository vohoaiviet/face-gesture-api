208  void AAMBody::AlignShape(AAM_Shape& pStartShape, const AAM_Shape& pDetShape, const AAM_Shape& pMeanShape)
209  {
210  AAM_Shape baseShape, alignedShape;
211  baseShape.resize(2);
212  alignedShape.resize(2);
213 
214  double meanCenter = (pMeanShape.MinY() + pMeanShape.MaxY()) * 0.5;
215  baseShape[0].x = pMeanShape.MinX();
216  baseShape[0].y = meanCenter;
217  baseShape[1].x = pMeanShape.MaxX();
218  baseShape[1].y = meanCenter;
219 
220  double yMean = (pDetShape[1].y + pDetShape[0].y) * 0.5;
221  alignedShape[0].x = pDetShape[0].x;
222  alignedShape[0].y = yMean;
223  alignedShape[1].x = pDetShape[1].x;
224  alignedShape[1].y = yMean;
225 
226  double a, b, tx, ty;
227  baseShape.AlignTransformation(alignedShape, a, b, tx, ty);
228  pStartShape = pMeanShape;
229  pStartShape.TransformPose(a, b, tx, ty);

233  void AAMBody::AdjustShape(AAM_Shape& pShape)
234  {
235  double xMin = pShape[0].x;
236  double yMin = pShape[0].y;
237  double xMax = pShape[1].x;
238  double yMax = pShape[1].y;
239 
240  double NewHeight = param_->shrink.y * (yMax - yMin);
241  double NewWidth = param_->shrink.x * (xMax - xMin);
242  double yMean = (yMin + yMax) * 0.5;
243  double xMean = (xMin + xMax) * 0.5;
244 
245  yMean += param_->shift.y * (yMax - yMin); // move face down
246  xMean += param_->shift.x * (xMax - xMin); // move face down
247 
248  pShape[0].x = xMean - 0.5 * NewWidth;
249  pShape[0].y = yMean - 0.5 * NewHeight;
250  pShape[1].x = xMean + 0.5 * NewWidth;
251  pShape[1].y = yMean + 0.5 * NewHeight;
252  }
